# On Any Push
on:
  push:

# On Push to specific relative path
on:
  push:
    paths:
      - 'projects/proxmox-test1/**'


# On a schedule
on:
  schedule:
    - cron: '0 01 * * 4'

# On Push to specific Branch
on:
  push:
    branches:
      - 'main'

# Can use multiple triggers together
on:
  push:
    branches:
      - 'main'
    paths:
      - 'epichouse.web/**'
